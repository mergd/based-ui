## Usage

```tsx title="Imports"
import {
	Sheet,
	SheetClose,
	SheetContent,
	SheetDescription,
	SheetFooter,
	SheetHeader,
	SheetTitle,
	SheetTrigger,
} from "@/components/ui/sheet"
```

```tsx title="Anatomy"
<Sheet>
	<SheetTrigger />
	<SheetContent>
		<SheetClose />
		<SheetHeader>
			<SheetTitle />
			<SheetDescription />
		</SheetHeader>
		<SheetFooter>
			<SheetClose />
		</SheetFooter>
	</SheetContent>
</Sheet>
```
