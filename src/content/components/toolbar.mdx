## Usage

```tsx title="Imports"
import {
	Toolbar,
	ToolbarButton,
	ToolbarGroup,
	ToolbarInput,
	ToolbarLink,
	ToolbarSeparator,
} from "@/components/ui/toolbar"
```

```tsx title="Anatomy"
<Toolbar>
  <ToolbarButton />
  <ToolbarLink />
  <ToolbarSeparator />
  <ToolbarGroup>
    <ToolbarButton />
    <ToolbarButton />
  <ToolbarGroup />
  <ToolbarInput />
</Toolbar>
```
