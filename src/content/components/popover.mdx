## Usage

```tsx title="Imports"
import {
	Popover,
	PopoverClose,
	PopoverContent,
	PopoverDescription,
	PopoverFooter,
	PopoverHeader,
	PopoverTitle,
	PopoverTrigger,
} from "@/components/ui/popover"
```

```tsx title="Anatomy"
<Popover>
	<PopoverTrigger />
	<PopoverContent>
		<PopoverHeader>
			<PopoverTitle />
			<PopoverDescription />
		</PopoverHeader>
		<PopoverFooter>
			<PopoverClose />
		</PopoverFooter>
	</PopoverContent>
</Popover>
```
