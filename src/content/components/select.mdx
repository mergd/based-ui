## Usage

```tsx title="Imports"
import {
	Select,
	SelectContent,
	SelectGroup,
	SelectItem,
	SelectTrigger,
	SelectValue,
} from "@/components/ui/select"
```

```tsx title="Anatomy"
<Select>
	<SelectTrigger>
		<SelectValue />
	</SelectTrigger>
	<SelectContent>
		<SelectGroup>
			<SelectGroupLabel />
			<SelectItem />
		</SelectGroup>
	</SelectContent>
</Select>
```
