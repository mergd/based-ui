## Usage

```tsx title="Imports"
import {
	AlertDialog,
	AlertDialogContent,
	AlertDialogDescription,
	AlertDialogFooter,
	AlertDialogHeader,
	AlertDialogTitle,
	AlertDialogTrigger,
} from "@/components/ui/alert-dialog"
```

```tsx title="Anatomy"
<AlertDialog>
	<AlertDialogTrigger />
	<AlertDialogContent>
		<AlertDialogHeader>
			<AlertDialogTitle />
			<AlertDialogDescription />
		</AlertDialogHeader>
		<AlertDialogFooter>
			<AlertDialogClose />
		</AlertDialogFooter>
	</AlertDialogContent>
</AlertDialog>
```
