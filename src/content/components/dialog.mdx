## Usage

```tsx title="Imports"
import {
	Dialog,
	DialogClose,
	DialogContent,
	DialogDescription,
	DialogFooter,
	DialogHeader,
	DialogTitle,
	DialogTrigger,
} from "@/components/ui/dialog"
```

```tsx title="Anatomy"
<Dialog>
	<DialogTrigger />
	<DialogContent>
		<DialogHeader>
			<DialogTitle />
			<DialogDescription />
		</DialogHeader>
		<DialogFooter>
			<DialogClose />
		</DialogFooter>
	</DialogContent>
</Dialog>
```
